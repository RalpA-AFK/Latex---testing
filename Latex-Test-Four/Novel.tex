% Slide 6: Story Breakdown
\begin{frame}{Story Breakdown}
  \noindent\textbf{Setting:} American suburbs. Rural roads, diners, motel rooms. \\[0.4em]
  \noindent\textbf{Characters:} James, who has decided he is a psychopath. Alyssa, who does not care what anyone thinks. \\[0.4em]
  \noindent\textbf{Inciting Incident:} James decides he wants to kill someone. He picks Alyssa. \\[0.4em]
  \noindent\textbf{Rising Action:} They run. They break into empty houses and eat whatever is left in the cabinets. James had every intention of killing her. The longer they drive, the less certain he is about that. \\[0.4em]
  \noindent\textbf{Satisfactory End:} James is shot. His last words are ``I love you.'' Alyssa carves his name into her arm.
\end{frame}

  \begin{columns}[T]
